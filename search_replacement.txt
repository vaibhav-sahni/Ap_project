        // Simple Search Field (no icon)
        final String placeholder = "Search by name...";
        
        JPanel searchPanel = new JPanel(new BorderLayout());
        searchPanel.setOpaque(true);
        boolean dark = FlatLaf.isLafDark();
        searchPanel.setBackground(dark ? Color.decode("#2A2A2A") : Color.decode("#F8F9FA"));
        searchPanel.setBorder(BorderFactory.createCompoundBorder(
            BorderFactory.createLineBorder(dark ? Color.decode("#404040") : Color.decode("#E2E8F0"), 1, true),
            BorderFactory.createEmptyBorder(10, 14, 10, 14)
        ));

        searchField = new JTextField();
        searchField.setBorder(BorderFactory.createEmptyBorder(0, 0, 0, 0));
        searchField.setFont(new Font(Font.SANS_SERIF, Font.PLAIN, 13));
        searchField.setOpaque(false);
        searchField.setText(placeholder);
        searchField.setForeground(dark ? Color.decode("#666666") : Color.decode("#94A3B8"));

        searchField.addFocusListener(new FocusAdapter() {
            @Override
            public void focusGained(FocusEvent e) {
                if (searchField.getText().equals(placeholder)) {
                    searchField.setText("");
                    boolean dark = FlatLaf.isLafDark();
                    searchField.setForeground(dark ? Color.decode("#EAEAEA") : Color.decode("#1E1E1E"));
                }
            }

            @Override
            public void focusLost(FocusEvent e) {
                if (searchField.getText().trim().isEmpty()) {
                    boolean dark = FlatLaf.isLafDark();
                    searchField.setForeground(dark ? Color.decode("#666666") : Color.decode("#94A3B8"));
                    searchField.setText(placeholder);
                }
            }
        });

        searchField.addKeyListener(new KeyAdapter() {
            @Override
            public void keyReleased(KeyEvent e) {
                applySearchFilter();
            }
        });

        searchPanel.add(searchField, BorderLayout.CENTER);
        bar.add(searchPanel, "h 40!, growx");
